{
	=<p>https</p> -> scheme

    =<p>fh-kufstein.ac.at</p> -> host
    =<p>johnny\:secret123@fh-kufstein.ac.at\:8080</p> -> authority
    =<p>/web/basics/script.php</p> -> path
    =<p>q1</p> -> fragment
    =<p>sort\=asc&amp;page\=3</p> -> query
}
